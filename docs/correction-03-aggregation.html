Pré-requis Jupyter

import requests
import json

ES_URL = "http://elasticsearch:9200"
INDEX = "shakespeare_extended"

def run(query):
    resp = requests.get(f"{ES_URL}/{INDEX}/_search", json=query)
    data = resp.json()
    print(json.dumps(data, indent=2, ensure_ascii=False))
    return data
